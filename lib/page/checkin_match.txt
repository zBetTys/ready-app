  Future<void> _loadFaceDataFromFirebase() async {
    if (_targetUserId == null || _targetUserId!.isEmpty) {
      _addDebugLog('⚠️ _targetUserId เป็น null หรือว่าง');
      return;
    }

    setState(() {
      _dataLoading = true;
      _statusMessage = '🔄 กำลังโหลดข้อมูลใบหน้า...';
    });

    try {
      _faceProfiles.clear();
      _faceEmbeddings.clear();
      _primaryProfile = null;

      _addDebugLog('\n📥 ===== โหลด Face Data =====');
      _addDebugLog('👤 User ID: $_targetUserId');

      final userRef = _firestore.collection('users').doc(_targetUserId);

      // ========== โหลด Face Profiles ==========
      _addDebugLog('\n📁 กำลังโหลด face_profiles...');

      try {
        final profilesSnapshot =
            await userRef.collection('face_profiles').get();

        _addDebugLog('📊 พบ ${profilesSnapshot.docs.length} รายการ');

        for (var doc in profilesSnapshot.docs) {
          final data = doc.data();
          _addDebugLog('   📌 Profile ID: ${doc.id}');

          List<double>? meanEmbedding;

          if (data.containsKey('mean_embedding')) {
            meanEmbedding = _tryParseEmbedding(data['mean_embedding']);
            _addDebugLog('      ✅ พบ mean_embedding');
          } else if (data.containsKey('embedding')) {
            meanEmbedding = _tryParseEmbedding(data['embedding']);
            _addDebugLog('      ✅ พบ embedding');
          }

          if (meanEmbedding != null) {
            _addDebugLog('      📏 dimension: ${meanEmbedding.length}');

            _faceProfiles.add({
              'id': doc.id,
              'mean_embedding': meanEmbedding,
              'confidence':
                  data['face_profile_confidence'] ?? data['confidence'] ?? 0.86,
              'created_at': data['createdAt'] ?? data['created_at'],
            });

            if (_primaryProfile == null) {
              _primaryProfile = {
                'id': doc.id,
                'mean_embedding': meanEmbedding,
                'confidence': data['face_profile_confidence'] ?? 0.86,
              };
            }
          }
        }
      } catch (e) {
        _addDebugLog('❌ Error loading profiles: $e');
      }

      // ========== โหลด Face Embeddings ==========
      _addDebugLog('\n📁 กำลังโหลด face_embeddings...');

      try {
        final embeddingsSnapshot =
            await userRef.collection('face_embeddings').limit(20).get();

        _addDebugLog('📊 พบ ${embeddingsSnapshot.docs.length} รายการ');

        for (var doc in embeddingsSnapshot.docs) {
          final data = doc.data();
          _addDebugLog('   📌 Embedding ID: ${doc.id}');

          List<double>? embedding;

          if (data.containsKey('embedding')) {
            embedding = _tryParseEmbedding(data['embedding']);
            _addDebugLog('      ✅ พบ embedding');
          } else if (data.containsKey('embedding_vector')) {
            embedding = _tryParseEmbedding(data['embedding_vector']);
            _addDebugLog('      ✅ พบ embedding_vector');
          } else if (data.containsKey('vector')) {
            embedding = _tryParseEmbedding(data['vector']);
            _addDebugLog('      ✅ พบ vector');
          }

          if (embedding != null) {
            _addDebugLog('      📏 dimension: ${embedding.length}');

            _faceEmbeddings.add({
              'id': doc.id,
              'profile_id': data['profile_id'] ?? '',
              'embedding': embedding,
              'quality': data['quality'] ?? data['quality_score'] ?? 0.7,
            });
          }
        }
      } catch (e) {
        _addDebugLog('❌ Error loading embeddings: $e');
      }

      final hasData = _faceProfiles.isNotEmpty || _faceEmbeddings.isNotEmpty;

      _addDebugLog('\n📊 สรุป:');
      _addDebugLog('   - Profiles: ${_faceProfiles.length}');
      _addDebugLog('   - Embeddings: ${_faceEmbeddings.length}');
      _addDebugLog('   - มีข้อมูล: $hasData');

      setState(() {
        _dataLoaded = hasData;
        _dataLoading = false;

        if (hasData) {
          _statusMessage = '✅ พร้อมเช็คชื่อ';
          _instructionMessage = 'วางใบหน้าในกรอบ';
        } else {
          _statusMessage = '⚠️ ไม่พบข้อมูลใบหน้า';
          _instructionMessage = 'กรุณาลงทะเบียนก่อน';
          _createMockData();
        }
      });
    } catch (e) {
      _addDebugLog('❌ Error loading face data: $e');
      setState(() {
        _dataLoaded = false;
        _dataLoading = false;
        _statusMessage = '❌ โหลดข้อมูลไม่สำเร็จ';
      });
      _createMockData();
    }
  }

  void _createMockData() {
    _addDebugLog('📁 สร้างข้อมูลจำลองตามรูป');

    final mockEmbedding =
        List.generate(192, (i) => (Random().nextDouble() * 2) - 1);
    final normalized = _l2Normalize(mockEmbedding);

    _faceProfiles.add({
      'id': 'mock_profile',
      'mean_embedding': normalized,
      'confidence': 0.86,
      'created_at': DateTime.now().toIso8601String(),
    });

    _primaryProfile = _faceProfiles.first;

    for (int i = 0; i < 3; i++) {
      final emb = List.generate(192, (j) => (Random().nextDouble() * 2) - 1);
      final normalizedEmb = _l2Normalize(emb);

      _faceEmbeddings.add({
        'id': 'mock_emb_$i',
        'profile_id': 'mock_profile',
        'embedding': normalizedEmb,
        'quality': 0.7 + (i * 0.05),
      });
    }

    _dataLoaded = true;
    _addDebugLog('✅ สร้างข้อมูลจำลอง: 1 profile, 3 embeddings');
  }

  List<double>? _tryParseEmbedding(dynamic value) {
    try {
      if (value == null) return null;

      if (value is List) {
        return List<double>.from(value.map((e) {
          if (e is int) return e.toDouble();
          if (e is double) return e;
          if (e is num) return e.toDouble();
          return 0.0;
        }));
      } else if (value is String) {
        try {
          final parsed = jsonDecode(value);
          if (parsed is List) {
            return List<double>.from(parsed.map((e) {
              if (e is num) return e.toDouble();
              return 0.0;
            }));
          }
        } catch (e) {
          _addDebugLog('⚠️ แปลง JSON ไม่สำเร็จ: $e');
        }
      }

      return null;
    } catch (e) {
      _addDebugLog('⚠️ Error parsing embedding: $e');
      return null;
    }
  }

  // ================ LOAD MOBILENET MODEL ================
  Future<void> _loadMobileFaceNetModel() async {
    if (_modelLoading) return;

    setState(() {
      _modelLoading = true;
      _statusMessage = '🔄 กำลังโหลดโมเดล AI...';
    });

    try {
      _addDebugLog('\n🧠 โหลดโมเดล MobileFaceNet');

      try {
        final modelFile = await rootBundle.load(MODEL_ASSET_PATH);
        _addDebugLog('✅ พบไฟล์โมเดล: ${modelFile.lengthInBytes} bytes');
      } catch (e) {
        _addDebugLog('❌ ไม่พบไฟล์โมเดล: $e');
        _modelError = 'ไม่พบไฟล์โมเดล';
        _modelLoaded = false;
        return;
      }

      final interpreterOptions = InterpreterOptions()
        ..threads = 4
        ..useNnApiForAndroid = true;

      _faceModel = await Interpreter.fromAsset(
        MODEL_ASSET_PATH,
        options: interpreterOptions,
      );

      if (_faceModel == null) throw Exception('โมเดลเป็น null');

      final outputShape = _faceModel!.getOutputTensor(0).shape;
      _embeddingDimension = 1;
      for (var dim in outputShape) {
        _embeddingDimension *= dim;
      }
      _addDebugLog('📊 Embedding dimension: $_embeddingDimension');

      await _testModelInference();

      _modelLoaded = true;
      _addDebugLog('✅ โหลดโมเดลสำเร็จ');
    } catch (e) {
      _addDebugLog('❌ โหลดโมเดลล้มเหลว: $e');
      _modelError = e.toString();
      _modelLoaded = false;
    } finally {
      if (mounted) setState(() => _modelLoading = false);
    }
  }

  Future<void> _testModelInference() async {
    if (_faceModel == null) return;

    try {
      final testInput = List.generate(
        1,
        (_) => List.generate(
          FACE_CROP_SIZE,
          (_) => List.generate(
            FACE_CROP_SIZE,
            (_) => List.filled(3, 0.0),
          ),
        ),
      );

      final outputShape = _faceModel!.getOutputTensor(0).shape;
      int outputSize = 1;
      for (var dim in outputShape) outputSize *= dim;

      final outputBuffer = List<double>.filled(outputSize, 0.0);
      final output = outputBuffer.reshape(outputShape);

      final stopwatch = Stopwatch()..start();
      _faceModel!.run(testInput, output);
      stopwatch.stop();

      _addDebugLog('✅ Inference test: ${stopwatch.elapsedMilliseconds}ms');
    } catch (e) {
      _addDebugLog('❌ Inference test failed: $e');
      rethrow;
    }
  }

  // ================ INITIALIZE CAMERA ================
  Future<void> _initializeCamera() async {
    try {
      _cameras = await availableCameras();
      if (_cameras!.isEmpty) throw Exception('ไม่พบกล้อง');

      final frontCamera = _cameras!.firstWhere(
        (camera) => camera.lensDirection == CameraLensDirection.front,
        orElse: () => _cameras!.first,
      );

      _cameraController = CameraController(
        frontCamera,
        ResolutionPreset.medium,
        enableAudio: false,
        imageFormatGroup: ImageFormatGroup.nv21,
      );

      await _cameraController!.initialize();
      _addDebugLog('✅ กล้องพร้อม: ${frontCamera.name}');
    } catch (e) {
      _addDebugLog('❌ Camera error: $e');
      rethrow;
    }

  // ================ FACE DETECTION ================
  void _startFaceDetection() {
    _detectionTimer?.cancel();

    _detectionTimer =
        Timer.periodic(const Duration(milliseconds: 300), (timer) async {
      if (!_isCameraReady || _isMatching || _matchSuccess || _matchFailed)
        return;
      if (_cameraController == null || !_cameraController!.value.isInitialized)
        return;
      if (_isProcessing) return;
      if (_lastMatchAttempt != null) {
        final timeSinceLast = DateTime.now().difference(_lastMatchAttempt!);
        if (timeSinceLast.inSeconds < 2) return;
      }

      try {
        _isProcessing = true;

        final imageFile = await _cameraController!.takePicture();
        _addDebugLog('📸 ถ่ายภาพ: ${imageFile.path}');

        final inputImage = InputImage.fromFilePath(imageFile.path);
        final faces = await _faceDetector!.processImage(inputImage);

        if (faces.isNotEmpty) {
          final face = faces.first;

          setState(() {
            _currentFace = face;
          });

          _faceHistory.add(face);
          if (_faceHistory.length > 5) _faceHistory.removeAt(0);

          final quality = _calculateFaceQuality(face);
          final stability = _calculateFaceStability();
          final aligned = _checkFaceAlignment(face);

          setState(() {
            _faceQuality = quality;
            _faceStability = stability;
            _isFaceAligned = aligned;
          });

          _updateGuidance(face, quality, stability, aligned);

          if (_shouldMatch(quality, stability, aligned)) {
            setState(() => _stableFrameCount++);

            if (_stableFrameCount >= REQUIRED_STABLE_FRAMES && !_isMatching) {
              _lastMatchAttempt = DateTime.now();
              await _performFaceMatching(imageFile.path, face);

              try {
                if (await File(imageFile.path).exists()) {
                  await File(imageFile.path).delete();
                  _addDebugLog('🗑️ ลบไฟล์: ${imageFile.path}');
                }
              } catch (e) {
                _addDebugLog('⚠️ ลบไฟล์ไม่สำเร็จ: $e');
              }
            }
          } else {
            setState(() => _stableFrameCount = 0);

            try {
              if (await File(imageFile.path).exists()) {
                await File(imageFile.path).delete();
              }
            } catch (_) {}
          }
        } else {
          setState(() {
            _currentFace = null;
            _stableFrameCount = 0;
          });
          _updateStatus('👤 ไม่พบใบหน้า', 'วางใบหน้าในกรอบ', '');

          try {
            if (await File(imageFile.path).exists()) {
              await File(imageFile.path).delete();
            }
          } catch (_) {}
        }

        _isProcessing = false;
      } catch (e) {
        _addDebugLog('Detection error: $e');
        _isProcessing = false;
      }
    });
  }

  double _calculateFaceQuality(Face face) {
    double score = 0.0;
    final bbox = face.boundingBox;
    final area = bbox.width * bbox.height;
    final screenArea = _screenWidth * _screenHeight;
    final areaRatio = area / screenArea;

    if (areaRatio >= 0.15 && areaRatio <= 0.30)
      score += 0.3;
    else if (areaRatio >= 0.10 && areaRatio <= 0.40)
      score += 0.2;
    else
      score += 0.1;

    score += _calculateCenterScore(bbox) * 0.25;

    final yaw = face.headEulerAngleY?.abs() ?? 0.0;
    final pitch = face.headEulerAngleX?.abs() ?? 0.0;
    final roll = face.headEulerAngleZ?.abs() ?? 0.0;

    if (yaw <= 15 && pitch <= 15 && roll <= 15)
      score += 0.25;
    else if (yaw <= 25 && pitch <= 20 && roll <= 20)
      score += 0.2;
    else
      score += 0.1;

    final leftEye = face.leftEyeOpenProbability ?? 0.5;
    final rightEye = face.rightEyeOpenProbability ?? 0.5;
    final eyeScore = (leftEye + rightEye) / 2;

    if (eyeScore >= 0.2)
      score += 0.2;
    else
      score += eyeScore * 0.2;

    return score.clamp(0.0, 1.0);
  }

  double _calculateCenterScore(Rect bbox) {
    final faceCenter =
        Offset(bbox.left + bbox.width / 2, bbox.top + bbox.height / 2);
    final screenCenter = Offset(_screenWidth / 2, _screenHeight / 2);
    final distance = (faceCenter - screenCenter).distance;
    final maxDistance = _screenWidth * 0.2;
    return max(0.0, 1.0 - (distance / maxDistance).clamp(0.0, 1.0));
  }

  double _calculateFaceStability() {
    if (_faceHistory.length < 2) return 1.0;
    double totalMovement = 0.0;
    for (int i = 1; i < _faceHistory.length; i++) {
      final prev = _faceHistory[i - 1];
      final curr = _faceHistory[i];
      final prevCenter = Offset(
          prev.boundingBox.left + prev.boundingBox.width / 2,
          prev.boundingBox.top + prev.boundingBox.height / 2);
      final currCenter = Offset(
          curr.boundingBox.left + curr.boundingBox.width / 2,
          curr.boundingBox.top + curr.boundingBox.height / 2);
      totalMovement += (currCenter - prevCenter).distance;
    }
    final avgMovement = totalMovement / (_faceHistory.length - 1);
    return max(0.0, 1.0 - (avgMovement / 15.0)).clamp(0.0, 1.0);
  }

  bool _checkFaceAlignment(Face face) {
    final yaw = face.headEulerAngleY?.abs() ?? 0.0;
    final pitch = face.headEulerAngleX?.abs() ?? 0.0;
    final roll = face.headEulerAngleZ?.abs() ?? 0.0;
    return yaw <= MAX_HEAD_YAW &&
        pitch <= MAX_HEAD_PITCH &&
        roll <= MAX_HEAD_ROLL;
  }

  bool _shouldMatch(double quality, double stability, bool aligned) {
    return aligned &&
        quality >= MIN_FACE_QUALITY &&
        stability >= MIN_FACE_STABILITY;
  }

  void _updateGuidance(
      Face face, double quality, double stability, bool aligned) {
    final bbox = face.boundingBox;
    final area = bbox.width * bbox.height;
    final screenArea = _screenWidth * _screenHeight;
    final areaRatio = area / screenArea;

    if (areaRatio > MAX_FACE_SIZE) {
      _updateStatus('📱 ถอยหลังหน่อย', 'ใบหน้าใกล้เกินไป', '');
      return;
    } else if (areaRatio < MIN_FACE_SIZE) {
      _updateStatus('📱 ขยับเข้ามาใกล้ๆ', 'ใบหน้าไกลเกินไป', '');
      return;
    }

    final centerScore = _calculateCenterScore(bbox);
    if (centerScore < 0.7) {
      _updateStatus('🎯 จัดใบหน้าให้อยู่กึ่งกลาง', 'ปรับตำแหน่ง', '');
      return;
    }

    if (!aligned) {
      _updateStatus('↔️ หันหน้าตรง', 'จัดใบหน้าให้ตรง', '');
      return;
    }

    if (quality < MIN_FACE_QUALITY) {
      _updateStatus(
          '📸 ปรับแสงและตำแหน่ง', 'คุณภาพ ${(quality * 100).toInt()}%', '');
      return;
    }

    if (stability < MIN_FACE_STABILITY) {
      _updateStatus(
          '🎯 อยู่นิ่งๆ', 'ความนิ่ง ${(stability * 100).toInt()}%', '');
      return;
    }

    _updateStatus('✅ พร้อมตรวจสอบ', 'ระบบกำลังตรวจสอบใบหน้า...', '');
  }

  Future<img.Image?> _cropAndPreprocessFace(String imagePath, Face face) async {
    try {
      _addDebugLog('📂 กำลังอ่านไฟล์: $imagePath');

      final file = File(imagePath);
      if (!await file.exists()) {
        _addDebugLog('❌ ไฟล์ไม่มีอยู่: $imagePath');
        return null;
      }

      final bytes = await file.readAsBytes();
      _addDebugLog('📊 ขนาดไฟล์: ${bytes.length} bytes');

      final originalImage = img.decodeImage(bytes);
      if (originalImage == null) {
        _addDebugLog('❌ ถอดรหัสรูปไม่สำเร็จ');
        return null;
      }

      _addDebugLog(
          '✅ อ่านรูปสำเร็จ: ${originalImage.width}x${originalImage.height}');

      final bbox = face.boundingBox;
      final previewSize = _cameraController!.value.previewSize!;
      final scaleX = originalImage.width / previewSize.height;
      final scaleY = originalImage.height / previewSize.width;

      double left = bbox.left * scaleX;
      double top = bbox.top * scaleY;
      double right = bbox.right * scaleX;
      double bottom = bbox.bottom * scaleY;

      if (_cameraController!.description.lensDirection ==
          CameraLensDirection.front) {
        final tempLeft = left;
        left = originalImage.width - right;
        right = originalImage.width - tempLeft;
      }

      final paddingX = (right - left) * FACE_PADDING_RATIO;
      final paddingY = (bottom - top) * FACE_PADDING_RATIO;

      left = (left - paddingX).clamp(0.0, originalImage.width.toDouble());
      top = (top - paddingY).clamp(0.0, originalImage.height.toDouble());
      right = (right + paddingX).clamp(0.0, originalImage.width.toDouble());
      bottom = (bottom + paddingY).clamp(0.0, originalImage.height.toDouble());

      final cropWidth = (right - left).toInt();
      final cropHeight = (bottom - top).toInt();

      if (cropWidth <= 0 || cropHeight <= 0) {
        _addDebugLog('❌ ขนาด crop ไม่ถูกต้อง: $cropWidth x $cropHeight');
        return null;
      }

      _addDebugLog(
          '✂️ Crop: ${left.toInt()},${top.toInt()} $cropWidth x $cropHeight');

      final cropped = img.copyCrop(originalImage,
          x: left.toInt(),
          y: top.toInt(),
          width: cropWidth,
          height: cropHeight);

      final resized = img.copyResize(cropped,
          width: FACE_CROP_SIZE, height: FACE_CROP_SIZE);

      _addDebugLog('✅ ประมวลผลใบหน้าสำเร็จ');
      return resized;
    } catch (e) {
      _addDebugLog('❌ Crop error: $e');
      return null;
    }
  }

  Future<List<double>> _extractEmbedding(img.Image faceImage) async {
    if (_faceModel == null) throw Exception('โมเดลไม่พร้อม');

    try {
      final input = List.generate(
          1,
          (_) => List.generate(
              FACE_CROP_SIZE,
              (_) =>
                  List.generate(FACE_CROP_SIZE, (_) => List.filled(3, 0.0))));

      for (int y = 0; y < FACE_CROP_SIZE; y++) {
        for (int x = 0; x < FACE_CROP_SIZE; x++) {
          final pixel = faceImage.getPixel(x, y);
          input[0][y][x][0] = (pixel.r / 127.5) - 1.0;
          input[0][y][x][1] = (pixel.g / 127.5) - 1.0;
          input[0][y][x][2] = (pixel.b / 127.5) - 1.0;
        }
      }

      final outputShape = _faceModel!.getOutputTensor(0).shape;
      int outputSize = 1;
      for (var dim in outputShape) outputSize *= dim;

      final outputBuffer = List<double>.filled(outputSize, 0.0);
      final output = outputBuffer.reshape(outputShape);

      _faceModel!.run(input, output);

      List<double> result = [];
      if (outputShape.length == 2)
        result = List<double>.from(output[0]);
      else if (outputShape.length == 1)
        result = List<double>.from(output);
      else
        result = _flattenOutput(output);

      return _l2Normalize(result);
    } catch (e) {
      _addDebugLog('❌ Extract embedding error: $e');
      rethrow;
    }
  }

  List<double> _flattenOutput(dynamic output) {
    List<double> result = [];
    void flatten(dynamic item) {
      if (item is List) {
        for (var subItem in item) flatten(subItem);
      } else if (item is double) result.add(item);
    }

    flatten(output);
    return result;
  }

  List<double> _l2Normalize(List<double> vector) {
    double norm = 0.0;
    for (final v in vector) norm += v * v;
    norm = sqrt(norm);
    if (norm < 1e-12) return vector;
    return vector.map((v) => v / norm).toList();
  }

  double _cosineSimilarity(List<double> a, List<double> b) {
    if (a.isEmpty || b.isEmpty || a.length != b.length) return 0.0;
    double dot = 0.0, normA = 0.0, normB = 0.0;
    for (int i = 0; i < a.length; i++) {
      dot += a[i] * b[i];
      normA += a[i] * a[i];
      normB += b[i] * b[i];
    }
    if (normA < 1e-12 || normB < 1e-12) return 0.0;
    final similarity = dot / (sqrt(normA) * sqrt(normB));
    return ((similarity + 1) / 2).clamp(0.0, 1.0);
  }

  // ================ MAIN FACE MATCHING ================
  Future<void> _performFaceMatching(String imagePath, Face face) async {
    if (_isMatching) return;

    setState(() {
      _isMatching = true;
      _statusMessage = '🔍 กำลังตรวจสอบ...';
    });

    try {
      _addDebugLog('\n🔍 เริ่มเปรียบเทียบใบหน้า');

      final processedImage = await _cropAndPreprocessFace(imagePath, face);
      if (processedImage == null) throw Exception('ไม่สามารถประมวลผลใบหน้าได้');

      final currentEmbedding = await _extractEmbedding(processedImage);
      _addDebugLog('📊 Current embedding: ${currentEmbedding.length} dim');

      List<Map<String, dynamic>> referenceVectors = [];

      for (var profile in _faceProfiles) {
        if (profile.containsKey('mean_embedding')) {
          referenceVectors.add({
            'type': 'profile_mean',
            'profile_id': profile['id'],
            'vector': profile['mean_embedding'],
            'confidence': profile['confidence'] ?? 0.8,
          });
        }
      }

      for (var emb in _faceEmbeddings) {
        referenceVectors.add({
          'type': 'single_embedding',
          'profile_id': emb['profile_id'],
          'vector': emb['embedding'],
          'confidence': emb['quality'] ?? 0.7,
        });
      }

      if (referenceVectors.isEmpty) throw Exception('ไม่มีข้อมูลเปรียบเทียบ');

      double bestScore = 0.0;
      String? bestProfileId;

      for (var ref in referenceVectors) {
        final refVector = ref['vector'] as List<double>;
        if (refVector.length == currentEmbedding.length) {
          final similarity = _cosineSimilarity(currentEmbedding, refVector);
          _addDebugLog(
              '   📍 ${ref['type']}: ${(similarity * 100).toStringAsFixed(1)}%');
          if (similarity > bestScore) {
            bestScore = similarity;
            bestProfileId = ref['profile_id'];
          }
        }
      }

      _addDebugLog('📊 Best: ${(bestScore * 100).toStringAsFixed(1)}%');

      setState(() => _similarityScore = bestScore);

      if (bestScore >= MATCH_THRESHOLD) {
        _addDebugLog('✅✅✅ MATCH!');
        await _handleMatchSuccess(bestProfileId, bestScore);
      } else {
        _addDebugLog('❌❌❌ NO MATCH');
        await _handleMatchFailure(bestScore);
      }
    } catch (e) {
      _addDebugLog('❌ Matching error: $e');
      _updateStatus('❌ ตรวจสอบไม่สำเร็จ', 'กรุณาลองใหม่', '');
    } finally {
      if (mounted) setState(() => _isMatching = false);
    }
  }

  Future<void> _handleMatchSuccess(String? profileId, double score) async {
    setState(() {
      _matchSuccess = true;
      _matchedProfileId = profileId;
      _statusMessage = '✅ เช็คชื่อสำเร็จ!';
      _instructionMessage = 'ยินดีต้อนรับ ${_userData?['firstName'] ?? ''}';
    });

    _showSuccessAnimation = true;
    _successController.forward();

    // ✅ บันทึกข้อมูล checkin พร้อม educationLevel, year, department
    await _saveCheckinRecord(score, profileId);

    await Future.delayed(const Duration(seconds: 3));

    if (mounted) Navigator.pop(context, {'success': true, 'score': score});
  }

  Future<void> _handleMatchFailure(double score) async {
    setState(() {
      _matchFailed = true;
      _statusMessage = '❌ ไม่ตรงกับข้อมูล';
      _instructionMessage = 'ความคล้าย ${(score * 100).toStringAsFixed(1)}%';
    });

    _showFailAnimation = true;
    _failController.forward();

    _cooldownTimer?.cancel();
    _cooldownTimer = Timer(const Duration(seconds: 3), () {
      if (mounted)
        setState(() {
          _matchFailed = false;
          _showFailAnimation = false;
          _failController.reset();
        });
    });
  }

      _addDebugLog('✅ อัปเดต users collection สำเร็จ');
    } catch (e) {
      _addDebugLog('❌ Error saving checkin: $e');

      // ✅ Fallback: ลองบันทึกเฉพาะข้อมูลสำคัญ
      try {
        _addDebugLog('🔄 พยายามบันทึกอีกครั้งแบบ minimal...');

        final fallbackData = {
          'checkin_id': _uuid.v4(),
          'user_id': _targetUserId,
          'checked_by': _currentUser!.uid,
          'timestamp': FieldValue.serverTimestamp(),
          'similarity_score': score,
          'is_match': true,
          'education_level': _userData?['educationLevel'] ?? 'ป.ตรี',
          'year': _userData?['year'] ?? '4',
          'department': _userData?['department'] ?? 'สาขาวิศวกรรมซอฟต์แวร์',
        };

        await _firestore.collection('checkins').doc().set(fallbackData);
        _addDebugLog('✅ บันทึก fallback สำเร็จ');
      } catch (fallbackError) {
        _addDebugLog('❌ Fallback ก็ล้มเหลว: $fallbackError');
      }
    }
  }

  void _updateStatus(String status, String instruction, String detail) {
    if (mounted)
      setState(() {
        _statusMessage = status;
        _instructionMessage = instruction;
        _detailMessage = detail;
      });
  }

  Color _getQualityColor(double quality) {
    if (quality >= 0.8) return Colors.green;
    if (quality >= 0.6) return Colors.lightGreen;
    if (quality >= 0.4) return Colors.orange;
    return Colors.red;
  }

  Color _getMatchColor(double score) {
    if (score >= MATCH_THRESHOLD) return Colors.green;
    if (score >= MATCH_THRESHOLD - 0.15) return Colors.orange;
    return Colors.red;
  }
